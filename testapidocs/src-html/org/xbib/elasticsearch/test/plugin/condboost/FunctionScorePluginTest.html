001package org.xbib.elasticsearch.test.plugin.condboost;
002
003import org.elasticsearch.action.search.SearchRequest;
004import org.elasticsearch.action.search.SearchResponse;
005import org.elasticsearch.action.search.SearchType;
006import org.elasticsearch.common.Priority;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.search.SearchHits;
009import org.junit.Test;
010import org.xbib.elasticsearch.index.query.functionscore.condboost.CondBoostFactorFunction;
011import org.xbib.elasticsearch.index.query.functionscore.condboost.CondBoostFactorFunctionBuilder;
012import org.xbib.elasticsearch.plugin.condboost.CondBoostPlugin;
013import org.xbib.elasticsearch.test.AbstractNodeTestHelper;
014
015import static org.elasticsearch.client.Requests.indexRequest;
016import static org.elasticsearch.client.Requests.searchRequest;
017import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
018import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
019import static org.elasticsearch.index.query.QueryBuilders.functionScoreQuery;
020import static org.elasticsearch.index.query.QueryBuilders.termQuery;
021import static org.elasticsearch.search.builder.SearchSourceBuilder.searchSource;
022import static org.junit.Assert.assertEquals;
023
024public class FunctionScorePluginTest extends AbstractNodeTestHelper {
025
026    @Override
027    protected Settings getNodeSettings() {
028        return settingsBuilder()
029                .put("plugin.types", CondBoostPlugin.class.getName())
030                .put(super.getNodeSettings())
031                .build();
032    }
033
034    @Test
035    public void testPlugin() throws Exception {
036        client().admin()
037                .indices()
038                .prepareCreate("test")
039                .execute().actionGet();
040
041        client().admin().cluster().prepareHealth()
042                .setWaitForEvents(Priority.LANGUID).setWaitForYellowStatus().execute().actionGet();
043
044        client().index(
045                indexRequest("test").type("products").id("1")
046                        .source(jsonBuilder().startObject()
047                                .field("content", "foo bar")
048                                .field("product", "product_name_1")
049                                .endObject())).actionGet();
050        client().index(
051                indexRequest("test").type("products").id("2")
052                        .source(jsonBuilder().startObject()
053                                .field("content", "foo bar")
054                                .field("product", "product_name_2")
055                                .endObject())).actionGet();
056        client().index(
057                indexRequest("test").type("products").id("3")
058                        .source(jsonBuilder().startObject()
059                                .field("content", "foo bar")
060                                .field("product", "product_name_3")
061                                .endObject())).actionGet();
062
063        client().index(
064                indexRequest("test").type("products").id("4")
065                        .source(jsonBuilder().startObject()
066                                .field("content", "foo bar")
067                                .field("product", "product_name_1")
068                                .field("user", "user_name_1")
069                                .endObject())).actionGet();
070        client().index(
071                indexRequest("test").type("products").id("5")
072                        .source(jsonBuilder().startObject()
073                                .field("content", "foo bar")
074                                .field("product", "product_name_2")
075                                .field("user", "user_name_1")
076                                .endObject())).actionGet();
077        client().index(
078                indexRequest("test").type("products").id("6")
079                        .source(jsonBuilder().startObject()
080                                .field("content", "foo bar")
081                                .field("product", "product_name_3")
082                                .field("user", "user_name_1")
083                                .endObject())).actionGet();
084
085        client().admin().indices().prepareRefresh().execute().actionGet();
086
087        CondBoostFactorFunctionBuilder cbfb = new CondBoostFactorFunctionBuilder()
088                .factor(1.0f)
089                .modifier(CondBoostFactorFunction.Modifier.NONE)
090                .condBoost("product", "product_name_1", 2.0f)
091                .condBoost("user", "user_name_1", 3.0f);
092
093        SearchRequest searchRequest = searchRequest()
094                .searchType(SearchType.QUERY_THEN_FETCH)
095                .source(searchSource()
096                        .explain(false)
097                        .query(functionScoreQuery(termQuery("content", "foo")).add(cbfb)));
098
099        SearchResponse sr = client().search(searchRequest).actionGet();
100        SearchHits sh = sr.getHits();
101
102        /*for (int i = 0; i < sh.hits().length; i++) {
103            System.err.println( sh.getAt(i).getId() + " " + sh.getAt(i).getScore() + " -->" + sh.getAt(i).getSource());
104        }*/
105
106        assertEquals(sh.hits().length, 6);
107        assertEquals(sh.getAt(0).getId(), "4");
108        assertEquals(sh.getAt(1).getId(), "5");
109        assertEquals(sh.getAt(2).getId(), "6");
110        assertEquals(sh.getAt(3).getId(), "1");
111    }
112
113}
