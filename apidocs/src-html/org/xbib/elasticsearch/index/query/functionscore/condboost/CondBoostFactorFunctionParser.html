002package org.xbib.elasticsearch.index.query.functionscore.condboost;
003
004import org.elasticsearch.ElasticsearchException;
005import org.elasticsearch.common.lucene.search.function.ScoreFunction;
006import org.elasticsearch.common.xcontent.XContentParser;
007import org.elasticsearch.index.mapper.FieldMapper;
008import org.elasticsearch.index.query.QueryParseContext;
009import org.elasticsearch.index.query.QueryParsingException;
010import org.elasticsearch.index.query.functionscore.ScoreFunctionParser;
011import org.elasticsearch.search.internal.SearchContext;
012
013import java.io.IOException;
014import java.util.List;
015import java.util.Locale;
016
017import static org.elasticsearch.common.collect.Lists.newLinkedList;
018
019/**
020 * Parses out a function_score function that looks like:
021 *
022 * <pre>
023 *     {
024 *         "cond_boost": {
025 *             "value" : 1.0,
026 *             "factor" : 1.0,
027 *             "modifier" : "NONE",
028 *             "cond" : [
029 *               {
030 *                 "product" : "product_name_1",
031 *                 "value" : 5.0
032 *               },
033 *               {
034 *                 "user" : "user_1",
035 *                 "value" : 10.0
036 *               }
037 *             ]
038 *         }
039 *     }
040 * </pre>
041 */
042public class CondBoostFactorFunctionParser implements ScoreFunctionParser {
043
044    public static String[] NAMES = { "cond_boost", "condBoost" };
045
046    @Override
047    public String[] getNames() {
048        return NAMES;
049    }
050
051    @Override
052    public ScoreFunction parse(QueryParseContext parseContext, XContentParser parser) throws IOException, QueryParsingException {
053        String currentFieldName = null;
054        List<CondBoostEntry> condArray = newLinkedList();
055        float defaultBoost = 1.0f;
056        float boostFactor = 1.0f;
057        CondBoostFactorFunction.Modifier modifier = CondBoostFactorFunction.Modifier.NONE;
058        XContentParser.Token token;
059        while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
060            if (token == XContentParser.Token.FIELD_NAME) {
061                currentFieldName = parser.currentName();
062            } else if (token == XContentParser.Token.START_ARRAY) {
063                condArray = parseCondArray(parseContext, parser, currentFieldName);
064            } else if (token.isValue()) {
065                switch (currentFieldName) {
066                    case "value":
067                        defaultBoost = parser.floatValue();
068                        break;
069                    case "factor":
070                        boostFactor = parser.floatValue();
071                        break;
072                    case "modifier":
073                        modifier = CondBoostFactorFunction.Modifier.valueOf(parser.text().toUpperCase(Locale.ROOT));
074                        break;
075                    default:
076                        throw new QueryParsingException(parseContext.index(), NAMES[0] + " query does not support [" + currentFieldName + "]");
077                }
078            }
079        }
080        return new CondBoostFactorFunction(condArray, defaultBoost, boostFactor, modifier);
081    }
082
083    private List<CondBoostEntry> parseCondArray(QueryParseContext parseContext, XContentParser parser, String currentFieldName) throws IOException {
084        XContentParser.Token token;
085        List<CondBoostEntry> condArray = newLinkedList();
086        while ((token = parser.nextToken()) != XContentParser.Token.END_ARRAY) {
087            if (token != XContentParser.Token.START_OBJECT) {
088                throw new QueryParsingException(parseContext.index(), "malformed query, expected a "
089                        + XContentParser.Token.START_OBJECT + " while parsing cond boost array, but got a " + token);
090            } else {
091                CondBoostEntry entry = new CondBoostEntry();
092                while ((token = parser.nextToken()) != XContentParser.Token.END_OBJECT) {
093                    if (token == XContentParser.Token.FIELD_NAME) {
094                        currentFieldName = parser.currentName();
095                    } else {
096                        if ("value".equals(currentFieldName)) {
097                            entry.boost = parser.floatValue();
098                        } else {
099                            entry.fieldName = currentFieldName;
100                            entry.fieldValue = parser.text();
101                            // compute IndexFieldData from currentFieldName
102                            SearchContext searchContext = SearchContext.current();
103                            FieldMapper mapper = searchContext.mapperService().smartNameFieldMapper(currentFieldName);
104                            if (mapper == null) {
105                                throw new ElasticsearchException("unable to find a field mapper for field [" + currentFieldName + "]");
106                            }
107                            entry.ifd = searchContext.fieldData().getForField(mapper);
108                        }
109                    }
110                }
111                condArray.add(entry);
112            }
113        }
114        return condArray;
115    }
116
117}
