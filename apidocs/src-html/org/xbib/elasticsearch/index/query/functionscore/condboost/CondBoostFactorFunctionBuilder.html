001package org.xbib.elasticsearch.index.query.functionscore.condboost;
002
003import org.elasticsearch.common.xcontent.XContentBuilder;
004import org.elasticsearch.index.query.functionscore.ScoreFunctionBuilder;
005
006import java.io.IOException;
007import java.util.List;
008import java.util.Locale;
009
010import static org.elasticsearch.common.collect.Lists.newLinkedList;
011
012public class CondBoostFactorFunctionBuilder implements ScoreFunctionBuilder {
013
014    private Float value = 1.0f;
015
016    private Float factor = 1.0f;
017
018    private CondBoostFactorFunction.Modifier modifier = CondBoostFactorFunction.Modifier.NONE;
019
020    private List<CondBoostEntry> entryList = newLinkedList();
021
022    @Override
023    public String getName() {
024        return CondBoostFactorFunctionParser.NAMES[0];
025    }
026
027    public CondBoostFactorFunctionBuilder condBoost(String fieldName,
028                                                    String fieldValue, float boost) {
029        CondBoostEntry entry = new CondBoostEntry();
030        entry.fieldName = fieldName;
031        entry.fieldValue = fieldValue;
032        entry.boost = boost;
033        entryList.add(entry);
034        return this;
035    }
036
037    public CondBoostFactorFunctionBuilder value(float boost) {
038        this.value = boost;
039        return this;
040    }
041
042    public CondBoostFactorFunctionBuilder factor(float boostFactor) {
043        this.factor = boostFactor;
044        return this;
045    }
046
047    public CondBoostFactorFunctionBuilder modifier(CondBoostFactorFunction.Modifier modifier) {
048        this.modifier = modifier;
049        return this;
050    }
051
052    @Override
053    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
054        builder.startObject(getName());
055        if (entryList != null) {
056            builder.startArray("cond");
057            for (CondBoostEntry entry : entryList) {
058                entry.toXContent(builder, params);
059            }
060            builder.endArray();
061        }
062        if (value != null) {
063            builder.field("value", value);
064        }
065        if (factor != null) {
066            builder.field("factor", factor);
067        }
068        if (modifier != null) {
069            builder.field("modifier", modifier.toString().toLowerCase(Locale.ROOT));
070        }
071        builder.endObject();
072        return builder;
073    }
074}
