001package org.xbib.elasticsearch.index.query.functionscore.condboost;
002
003import org.apache.lucene.index.AtomicReaderContext;
004import org.apache.lucene.search.Explanation;
005import org.apache.lucene.util.BytesRef;
006import org.elasticsearch.ElasticsearchException;
007import org.elasticsearch.common.lucene.search.function.CombineFunction;
008import org.elasticsearch.common.lucene.search.function.ScoreFunction;
009import org.elasticsearch.index.fielddata.BytesValues;
010
011import java.util.List;
012import java.util.Locale;
013
014import static org.elasticsearch.common.collect.Lists.newLinkedList;
015
016public class CondBoostFactorFunction extends ScoreFunction {
017
018    private AtomicReaderContext context;
019
020    private final List<CondBoostEntry> condBoostEntryList;
021
022    private final float boostFactor;
023
024    private final Modifier modifier;
025
026    private double defaultBoost;
027
028    private double boost;
029
030    private List<CondBoostEntry> appliedCondBoostEntryList;
031
032    public CondBoostFactorFunction(List<CondBoostEntry> condBoostEntryList,
033                                   float defaultBoost, float boostFactor, Modifier modifierType) {
034        super(CombineFunction.MULT);
035        this.condBoostEntryList = condBoostEntryList;
036        this.defaultBoost = defaultBoost;
037        this.boostFactor = boostFactor;
038        this.modifier = modifierType;
039    }
040
041    @Override
042    public void setNextReader(AtomicReaderContext context) {
043        this.context = context;
044    }
045
046    @Override
047    public double score(int docId, float subQueryScore) {
048        this.boost = defaultBoost;
049        this.appliedCondBoostEntryList = newLinkedList();
050        for (CondBoostEntry entry : condBoostEntryList) {
051            BytesValues values = entry.ifd.load(context).getBytesValues(true);
052            final int numValues = values.setDocument(docId);
053            for (int i = 0; i < numValues; i++) {
054                BytesRef value = values.nextValue();
055                if (entry.fieldValue.equals(value.utf8ToString())) {
056                    this.boost = this.boost * entry.boost; // multiply boosts by default
057                    appliedCondBoostEntryList.add(entry);
058                }
059            }
060        }
061        double val = boost * boostFactor;
062        double result = modifier.apply(val);
063        if (Double.isNaN(result) || Double.isInfinite(result)) {
064            throw new ElasticsearchException("result of field modification [" + modifier.toString() +
065                    "(" + val + ")] must be a number");
066        }
067        return result;
068    }
069
070    @Override
071    public Explanation explainScore(int docId, Explanation subQueryExpl) {
072        Explanation exp = new Explanation();
073        String modifierStr = modifier != null ? modifier.toString() : "";
074        double score = score(docId, subQueryExpl.getValue());
075        exp.setValue(CombineFunction.toFloat(score));
076        exp.setDescription("cond boost function: " +
077                modifierStr + "(" + appliedCondBoostEntryList + "->value " + boost + "  * factor=" + boostFactor + ")");
078        exp.addDetail(subQueryExpl);
079        return exp;
080    }
081
082    public enum Modifier {
083        NONE {
084            @Override
085            public double apply(double n) {
086                return n;
087            }
088        },
089        LOG {
090            @Override
091            public double apply(double n) {
092                return Math.log10(n);
093            }
094        },
095        LOG1P {
096            @Override
097            public double apply(double n) {
098                return Math.log10(n + 1);
099            }
100        },
101        LOG2P {
102            @Override
103            public double apply(double n) {
104                return Math.log10(n + 2);
105            }
106        },
107        LN {
108            @Override
109            public double apply(double n) {
110                return Math.log(n);
111            }
112        },
113        LN1P {
114            @Override
115            public double apply(double n) {
116                return Math.log1p(n);
117            }
118        },
119        LN2P {
120            @Override
121            public double apply(double n) {
122                return Math.log1p(n + 1);
123            }
124        },
125        SQUARE {
126            @Override
127            public double apply(double n) {
128                return Math.pow(n, 2);
129            }
130        },
131        SQRT {
132            @Override
133            public double apply(double n) {
134                return Math.sqrt(n);
135            }
136        },
137        RECIPROCAL {
138            @Override
139            public double apply(double n) {
140                return 1.0 / n;
141            }
142        };
143
144        public abstract double apply(double n);
145
146        @Override
147        public String toString() {
148            return super.toString().toLowerCase(Locale.ROOT);
149        }
150    }
151}
